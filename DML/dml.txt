// Departments and responsible entities
Table department {
  id INT PK
  name VARCHAR
}

Table responsible_entity {
  id INT PK
  name VARCHAR
}

Table department_entity {
  department_id INT PK
  entity_id INT PK
}

// Parks and areas
Table park {
  id INT PK
  name VARCHAR
  foundation_date DATE
}

Table department_park {
  department_id INT PK
  park_id INT PK
}

Table area {
  id INT PK
  name VARCHAR
  extent FLOAT
}

Table park_area {
  park_id INT PK
  area_id INT PK
}

// Species management
Table species {
  id INT PK
  type ENUM [note: 'vegetable, animal, mineral']
  common_name VARCHAR
  scientific_name VARCHAR
}

Table area_species {
  area_id INT PK
  species_id INT PK
  population INT
}

// Personnel base info
Table employee {
  id INT PK
  cedula INT UNIQUE
  name VARCHAR
  address VARCHAR
  phone VARCHAR
  mobile_phone VARCHAR
  salary DECIMAL
  role_type ENUM [note: '001-management, 002-vigilance, 003-conservation, 004-researcher']
}

// Personnel specialization tables
Table management_staff {
  employee_id INT PK
}

Table vigilance_staff {
  employee_id INT PK
}

Table conservation_staff {
  employee_id INT PK
  specialty VARCHAR [note: 'cleaning, path maintenance, etc.']
}

Table research_staff {
  employee_id INT PK
}

// Management staff activities
Table entrance {
  id INT PK
  park_id INT
  entrance_number INT
}

Table entrance_shift {
  id INT PK
  employee_id INT
  entrance_id INT
  begin TIMESTAMP
  end TIMESTAMP
}

// Vigilance staff equipment
Table vehicle {
  id INT PK
  type VARCHAR
  brand VARCHAR
}

Table vigilance_area {
  vigilance_id INT PK
  area_id INT PK
  vehicle_id INT
}

// Conservation staff assignments
Table conservation_area {
  conservation_id INT PK
  area_id INT PK
}

// Research projects
Table project {
  id INT PK
  name VARCHAR
  budget DECIMAL
  start_date DATE
  end_date DATE
}

Table project_researcher {
  project_id INT PK
  researcher_id INT PK
}

Table project_researcher_species {
  project_id INT PK
  researcher_id INT PK
  species_id INT PK
}

// Visitors
Table visitor {
  id INT PK
  cedula INT UNIQUE
  name VARCHAR
  address VARCHAR
  job VARCHAR
}

// Visit tracking
Table visit_log {
  id INT PK
  entrance_shift_id INT
  visit_time TIMESTAMP
  visitor_id INT
}

// Lodging
Table lodging {
  id INT PK
  park_id INT
  name VARCHAR
  capacity INT
  category VARCHAR
}

Table visitor_stay {
  id INT PK
  visitor_id INT
  lodging_id INT
  check_in DATE
  check_out DATE
}
